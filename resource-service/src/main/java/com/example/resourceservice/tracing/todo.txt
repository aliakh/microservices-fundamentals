    @Bean
    @LoadBalanced
    public RestTemplate restTemplate() {
        RestTemplate rt = new RestTemplate();
        List<ClientHttpRequestInterceptor> interceptors = new ArrayList<>(rt.getInterceptors());
        interceptors.add((request, body, execution) -> {
            String traceId = TraceContext.getOrCreateTraceId();
            request.getHeaders().set(TraceConstants.TRACE_ID_HEADER, traceId);
            return execution.execute(request, body);
        });
        rt.setInterceptors(interceptors);
        return rt;
    }

    @Bean
    public Consumer<Message<Long>> resourceProcessed() {
        return msg -> {
            Object headerVal = msg.getHeaders().get(TraceConstants.TRACE_ID_HEADER);
            if (headerVal instanceof String s && !s.isBlank()) {
                TraceContext.setTraceId(s);
            } else {
                TraceContext.getOrCreateTraceId();
            }
            try {
                resourceService.resourceProcessed(msg.getPayload());
            } finally {
                TraceContext.clear();
            }
        };
    }
}

    public void publishResourceCreated(Long resourceId) {
        String traceId = TraceContext.getOrCreateTraceId();
        streamBridge.send(
                "resourceCreated-out-0", // binding name
                MessageBuilder.withPayload(resourceId)
                        .setHeader(TraceConstants.TRACE_ID_HEADER, traceId)
                        .build()
        );
        System.out.println("Published event: ResourceCreated with ID=" + resourceId + ", traceId=" + traceId);
    }
