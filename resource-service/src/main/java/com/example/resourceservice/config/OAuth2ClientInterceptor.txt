package com.epam.learning.microservices.resource_service.interceptor;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpRequest;
import org.springframework.http.client.ClientHttpRequestExecution;
import org.springframework.http.client.ClientHttpRequestInterceptor;
import org.springframework.http.client.ClientHttpResponse;
import org.springframework.security.oauth2.client.OAuth2AuthorizeRequest;
import org.springframework.security.oauth2.client.OAuth2AuthorizedClient;
import org.springframework.security.oauth2.client.OAuth2AuthorizedClientManager;

import java.io.IOException;

/**
 * Interceptor that automatically adds OAuth2 Bearer tokens to outgoing HTTP requests.
 * Used for service-to-service authentication with client credentials grant.
 */
@Slf4j
@RequiredArgsConstructor
public class OAuth2ClientInterceptor implements ClientHttpRequestInterceptor {

    private final OAuth2AuthorizedClientManager authorizedClientManager;
    private final String clientRegistrationId;
    private final String principalName;

    @Override
    public ClientHttpResponse intercept(HttpRequest request, byte[] body, ClientHttpRequestExecution execution) throws IOException {

        // Build OAuth2 authorize request for client credentials
        OAuth2AuthorizeRequest authorizeRequest = OAuth2AuthorizeRequest
                .withClientRegistrationId(clientRegistrationId)
                .principal(principalName)
                .build();

        // Get the authorized client (this triggers token acquisition if needed)
        // The OAuth2AuthorizedClientManager handles token caching and refresh
        OAuth2AuthorizedClient authorizedClient = authorizedClientManager.authorize(authorizeRequest);

        // Add the access token to the Authorization header
        if (authorizedClient != null) {
            String tokenValue = authorizedClient.getAccessToken().getTokenValue();
            request.getHeaders().setBearerAuth(tokenValue);
            log.debug("Added OAuth2 Bearer token to request for client '{}'", clientRegistrationId);
        }

        // Continue with the request
        return execution.execute(request, body);
    }
}
