	// OAuth2 Client Manager - manages token acquisition and caching for service-to-service calls
	@Bean
	public OAuth2AuthorizedClientManager authorizedClientManager(
            ClientRegistrationRepository clientRegistrationRepository, OAuth2AuthorizedClientService authorizedClientService) {

		OAuth2AuthorizedClientProvider authorizedClientProvider = OAuth2AuthorizedClientProviderBuilder.builder()
						.clientCredentials()
						.build();

		AuthorizedClientServiceOAuth2AuthorizedClientManager authorizedClientManager =
				new AuthorizedClientServiceOAuth2AuthorizedClientManager(clientRegistrationRepository, authorizedClientService);
		authorizedClientManager.setAuthorizedClientProvider(authorizedClientProvider);

		return authorizedClientManager;
	}

	@Bean
	public RestClient restClient(OAuth2ClientInterceptor oauth2Interceptor, RestTemplate restTemplate) {
		return RestClient.builder(restTemplate)
				.requestInterceptor(oauth2Interceptor)
				.build();
	}

    @Bean
    public OAuth2ClientInterceptor oauth2Interceptor(OAuth2AuthorizedClientManager authorizedClientManager) {
        return new OAuth2ClientInterceptor(authorizedClientManager, "keycloak", "resource-service");
    }
