package com.example.server.config;

import io.micrometer.tracing.Span;
import io.micrometer.tracing.Tracer;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import org.slf4j.MDC;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Configuration;
import org.springframework.lang.NonNull;
import org.springframework.web.servlet.HandlerInterceptor;
import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

import java.util.UUID;

@Configuration
public class TracingMvcInterceptor implements WebMvcConfigurer {

    private final Tracer tracer;
    private final String traceHeader;

    public TracingMvcInterceptor(Tracer tracer,
                                 @Value("${app.tracing.header:X-Trace-Id}") String traceHeader) {
        this.tracer = tracer;
        this.traceHeader = traceHeader;
    }

    @Override
    public void addInterceptors(InterceptorRegistry registry) {
        registry.addInterceptor(new HandlerInterceptor() {
            @Override
            public boolean preHandle(@NonNull HttpServletRequest request,
                                     @NonNull HttpServletResponse response,
                                     @NonNull Object handler) {
                String incoming = request.getHeader(traceHeader);
                String traceId = currentTraceId();
                if (incoming != null && !incoming.isBlank()) {
                    MDC.put("traceId", incoming);
                } else if (traceId == null) {
                    Span s = tracer.nextSpan().name("http:incoming").start();
                    tracer.withSpan(s);
                    traceId = s.context().traceId();
                }
                response.setHeader(traceHeader, (incoming != null && !incoming.isBlank()) ? incoming : (traceId != null ? traceId : UUID.randomUUID().toString()));
                return true;
            }
        });
    }

    private String currentTraceId() {
        Span span = tracer.currentSpan();
        return span != null ? span.context().traceId() : MDC.get("traceId");
    }
}
package com.example.server.web;

import com.example.server.kafka.KafkaProducerService;
import io.micrometer.tracing.Tracer;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.multipart.MultipartFile;

@RestController
public class FileController {
    private static final Logger log = LoggerFactory.getLogger(FileController.class);

    private final Tracer tracer;
    private final KafkaProducerService producer;
    private final String traceHeader;

    public FileController(Tracer tracer, KafkaProducerService producer,
                          @Value("${app.tracing.header:X-Trace-Id}") String traceHeader) {
        this.tracer = tracer;
        this.producer = producer;
        this.traceHeader = traceHeader;
    }

    @PostMapping("/files")
    public ResponseEntity<String> upload(@RequestPart("file") MultipartFile file,
                                         @RequestHeader(value = "X-Trace-Id", required = false) String traceIdHeader) throws Exception {
        var span = tracer.currentSpan();
        String traceId = (span != null) ? span.context().traceId() : (traceIdHeader != null ? traceIdHeader : "no-trace");
        log.info("Server received file: name={}, size={}, traceId={}", file.getOriginalFilename(), file.getSize(), traceId);

        // Publish Kafka event with trace id in headers
        producer.publishFileProcessedEvent(traceId, file.getOriginalFilename(), file.getSize());

        return ResponseEntity.ok("server received file '" + file.getOriginalFilename() + "', traceId=" + traceId);
    }

    @GetMapping("/ping")
    public String ping() {
        var span = tracer.currentSpan();
        String traceId = (span != null) ? span.context().traceId() : "no-trace";
        log.info("server ping, traceId={}", traceId);
        return "server ok (traceId=" + traceId + ")";
    }
}

package com.example.server.kafka;

import io.micrometer.tracing.Span;
import io.micrometer.tracing.Tracer;
import org.apache.kafka.clients.producer.ProducerRecord;
import org.apache.kafka.common.header.internals.RecordHeader;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.kafka.core.KafkaTemplate;
import org.springframework.stereotype.Service;

import java.nio.charset.StandardCharsets;

@Service
public class KafkaProducerService {
    private static final Logger log = LoggerFactory.getLogger(KafkaProducerService.class);

    private final KafkaTemplate<String, String> kafkaTemplate;
    private final String topic;
    private final Tracer tracer;
    private final String traceHeader;

    public KafkaProducerService(KafkaTemplate<String, String> kafkaTemplate,
                                Tracer tracer,
                                @Value("${app.kafka.topic:files.process}") String topic,
                                @Value("${app.tracing.header:X-Trace-Id}") String traceHeader) {
        this.kafkaTemplate = kafkaTemplate;
        this.tracer = tracer;
        this.topic = topic;
        this.traceHeader = traceHeader;
    }

    public void publishFileProcessedEvent(String traceId, String filename, long size) {
        String payload = "processed:" + filename + ":" + size;
        ProducerRecord<String, String> record = new ProducerRecord<>(topic, filename, payload);
        String toSendTrace = (traceId != null && !traceId.isBlank()) ? traceId : currentTraceId();
        if (toSendTrace != null) {
            record.headers().add(new RecordHeader(traceHeader, toSendTrace.getBytes(StandardCharsets.UTF_8)));
        }
        // Optional: include W3C traceparent for linkage (if you want)
        // record.headers().add(new RecordHeader("traceparent", ...));

        log.info("Publishing Kafka message for file={}, size={}, traceId={}", filename, size, toSendTrace);
        kafkaTemplate.send(record);
    }

    private String currentTraceId() {
        Span span = tracer.currentSpan();
        return span != null ? span.context().traceId() : null;
    }
}


package com.example.server.kafka;

import io.micrometer.tracing.Span;
import io.micrometer.tracing.Tracer;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.slf4j.MDC;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.kafka.annotation.KafkaListener;
import org.springframework.kafka.support.KafkaHeaders;
import org.springframework.messaging.handler.annotation.Header;
import org.springframework.stereotype.Component;

@Component
public class KafkaConsumerService {
    private static final Logger log = LoggerFactory.getLogger(KafkaConsumerService.class);

    private final Tracer tracer;
    private final String traceHeader;

    public KafkaConsumerService(Tracer tracer, @Value("${app.tracing.header:X-Trace-Id}") String traceHeader) {
        this.tracer = tracer;
        this.traceHeader = traceHeader;
    }

    @KafkaListener(topics = "${app.kafka.topic:files.process}")
    public void onMessage(String payload,
                          @Header(name = KafkaHeaders.RECEIVED_KEY, required = false) String key,
                          @Header(name = "X-Trace-Id", required = false) String traceIdHeader) {
        // Start a new span and bind MDC with the propagated trace id if present
        Span span = tracer.nextSpan().name("kafka:consume:files.process").start();
        try (Tracer.SpanInScope ws = tracer.withSpan(span)) {
            if (traceIdHeader != null && !traceIdHeader.isBlank()) {
                MDC.put("traceId", traceIdHeader);
            }
            log.info("Consumed Kafka message, key={}, payload='{}', traceId={}", key, payload,
                    (traceIdHeader != null ? traceIdHeader : span.context().traceId()));
        } finally {
            span.end();
            MDC.remove("traceId");
        }
    }
}

How trace ID is propagated


Incoming HTTP (client & server)

MVC interceptor (TracingMvcInterceptor) reads X-Trace-Id header if provided and ensures it’s in MDC and response header.
If missing, it ensures a trace exists (Micrometer/OTel), and returns X-Trace-Id back.



Outgoing HTTP (client → server)

OpenFeign RequestInterceptor injects X-Trace-Id header using the current span (Micrometer Tracer).
Additionally, Micrometer + OTel instrumentation automatically injects W3C trace headers (traceparent, tracestate) for end-to-end traces in Tempo.



Kafka

Producer adds X-Trace-Id as a Kafka message header.
Consumer reads X-Trace-Id header, sets it into MDC, and logs with trace correlation. It also starts a span for consumption to keep the trace continuity.



Logs

Logback pattern includes %X{traceId} and %X{spanId} so all logs are searchable by traceId in Loki.



Metrics & Traces

Metrics exposed at /actuator/prometheus and scraped by Prometheus.
Traces exported via OTLP gRPC to Tempo (management.otlp.tracing.endpoint).

Notes / Tips

If you prefer JSON logs for easier parsing, swap to logstash-logback-encoder and update the Promtail pipeline accordingly. For this demo, a simple pattern with MDC is sufficient.
This example uses Bitnami Kafka in KRaft mode (no Zookeeper) for simplicity.
The shared /logs volume avoids host-specific Podman log scraping issues and keeps Promtail config trivial.
To view end-to-end traces in Grafana (Tempo), use Explore → Tempo and filter by service.name (client-service or server-service). You can jump from Logs (Loki) to Traces (Tempo) using the traceId from the log line.
Multipart upload in Feign is done with Resource to avoid MultipartFile creation on the client side.

3. 🔗 Trace ID Propagation Model
3.1 Incoming HTTP

MVC interceptor extracts X-Trace-Id (if present)
If missing, Micrometer/OTel generates a new span → new trace ID
The trace ID is added to MDC (%X{traceId})
Response always returns X-Trace-Id


3.2 Outgoing Feign requests

Feign RequestInterceptor reads:

Tracer.currentSpan().traceId OR
MDC traceId


Injects into header:

X-Trace-Id: <trace-id>


Micrometer automatically injects:

traceparent: <W3C-format>
tracestate: ...


3.3 Kafka Producer
Each message includes headers:
X-Trace-Id: <trace-id>
traceparent (optional)


3.4 Kafka Consumer

Extract header
Put into MDC
Start new span
Logs contain the same traceId
