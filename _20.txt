Awesome—let’s wire this up end‑to‑end the way you asked:

Spring Boot 3.5.9 + Spring Cloud 2025.0.1
Two services (client → server) with WebMVC and OpenFeign
Full trace ID propagation over HTTP and Kafka
MDC correlation in logs
Micrometer Tracing with OTLP → Tempo
Metrics to Prometheus
Logs to Loki via Promtail (from shared log files/volume)
All running in Podman Compose (with Kafka, Grafana, Prometheus, Loki, Tempo, Promtail)

Below is a complete, ready-to-run example with full source.

Trace strategy

Use Micrometer Tracing (OTel bridge) to automatically create spans and inject W3C trace headers (traceparent, tracestate).
Additionally add your custom X-Trace-Id header on every request and Kafka message.
Enrich logs via MDC with traceId/spanId so you can search logs by just the trace ID in Grafana/Loki.
Prometheus scrapes /actuator/prometheus.
Tempo receives OTLP (gRPC) at tempo:4317.

1) Podman Compose (all infra + both apps)
podman-compose.yml

version: "3.9"

volumes:
  app-logs:
  grafana-data:
  prometheus-data:
  loki-data:
  tempo-data:
  kafka-data:

networks:
  tracing-net:

services:
  tempo:
    image: grafana/tempo:2.6.0
    command: [ "-config.file=/etc/tempo/tempo-config.yaml" ]
    networks: [ tracing-net ]
    volumes:
      - ./tempo/tempo-config.yaml:/etc/tempo/tempo-config.yaml:ro
      - tempo-data:/var/tempo
    ports:
      - "3200:3200"     # HTTP (query)
      - "4317:4317"     # OTLP gRPC

  loki:
    image: grafana/loki:2.9.8
    command: [ "-config.file=/etc/loki/loki-config.yaml" ]
    networks: [ tracing-net ]
    volumes:
      - ./loki/loki-config.yaml:/etc/loki/loki-config.yaml:ro
      - loki-data:/loki
    ports:
      - "3100:3100"

  promtail:
    image: grafana/promtail:2.9.8
    command: [ "-config.file=/etc/promtail/promtail-config.yaml" ]
    networks: [ tracing-net ]
    volumes:
      - ./promtail/promtail-config.yaml:/etc/promtail/promtail-config.yaml:ro
      - app-logs:/var/log/app:ro
    depends_on:
      - loki

  prometheus:
    image: prom/prometheus:v2.53.0
    networks: [ tracing-net ]
    volumes:
      - ./prometheus/prometheus.yml:/etc/prometheus/prometheus.yml:ro
      - prometheus-data:/prometheus
    ports:
      - "9090:9090"

  grafana:
    image: grafana/grafana:11.1.0
    networks: [ tracing-net ]
    ports:
      - "3000:3000"
    volumes:
      - grafana-data:/var/lib/grafana
      - ./grafana/provisioning:/etc/grafana/provisioning:ro
    environment:
      - GF_SECURITY_ADMIN_PASSWORD=admin
      - GF_SECURITY_ADMIN_USER=admin
    depends_on:
      - prometheus
      - tempo
      - loki

  kafka:
    image: bitnami/kafka:3.7
    networks: [ tracing-net ]
    environment:
      - KAFKA_ENABLE_KRAFT=yes
      - KAFKA_CFG_PROCESS_ROLES=controller,broker
      - KAFKA_CFG_NODE_ID=1
      - KAFKA_CFG_CONTROLLER_QUORUM_VOTERS=1@kafka:9093
      - KAFKA_CFG_LISTENERS=PLAINTEXT://:9092,CONTROLLER://:9093
      - KAFKA_CFG_ADVERTISED_LISTENERS=PLAINTEXT://kafka:9092
      - KAFKA_CFG_CONTROLLER_LISTENER_NAMES=CONTROLLER
      - KAFKA_CFG_INTER_BROKER_LISTENER_NAME=PLAINTEXT
      - KAFKA_CFG_AUTO_CREATE_TOPICS_ENABLE=true
    volumes:
      - kafka-data:/bitnami
    ports:
      - "9092:9092"

  server-service:
    build:
      context: ./server-service
    image: local/server-service:latest
    networks: [ tracing-net ]
    environment:
      - SPRING_PROFILES_ACTIVE=default
      - SPRING_KAFKA_BOOTSTRAP_SERVERS=kafka:9092
      - MANAGEMENT_OTLP_TRACING_ENDPOINT=http://tempo:4317
    depends_on:
      - kafka
      - tempo
    volumes:
      - app-logs:/logs
    ports:
      - "8082:8082"

  client-service:
    build:
      context: ./client-service
    image: local/client-service:latest
    networks: [ tracing-net ]
    environment:
      - SPRING_PROFILES_ACTIVE=default
      - SERVER_SERVICE_URL=http://server-service:8082
      - MANAGEMENT_OTLP_TRACING_ENDPOINT=http://tempo:4317
    depends_on:
      - server-service
    volumes:
      - app-logs:/logs
    ports:
      - "8081:8081"
``

Tempo config — tempo/tempo-config.yaml

server:
  http_listen_port: 3200

distributor:
  receivers:
    otlp:
      protocols:
        grpc:            # listen on 0.0.0.0:4317 for OTLP gRPC
        http:

storage:
  trace:
    backend: local
    block:
      bloom_filter_false_positive: 0.05
      index_downsample_bytes: 1000
      size: 1_000_000
    wal:
      path: /var/tempo/wal
    local:
      path: /var/tempo/traces
compactor:
  compaction:
    block_retention: 48h

Promtail config — promtail/promtail-config.yaml
(we collect logs from a shared volume that both apps write to)

server:
  http_listen_port: 9080
  grpc_listen_port: 0

positions:
  filename: /tmp/positions.yaml

clients:
  - url: http://loki:3100/loki/api/v1/push

scrape_configs:
  - job_name: spring-apps
    static_configs:
      - targets: [localhost]
        labels:
          job: spring
          __path__: /var/log/app/*.log

Prometheus config — prometheus/prometheus.yml

global:
  scrape_interval: 10s

scrape_configs:
  - job_name: 'client-service'
    metrics_path: /actuator/prometheus
    static_configs:
      - targets: [ 'client-service:8081' ]

  - job_name: 'server-service'
    metrics_path: /actuator/prometheus
    static_configs:
      - targets: [ 'server-service:8082' ]

Grafana datasources — grafana/provisioning/datasources/datasource.yml

apiVersion: 1
datasources:
  - name: Prometheus
    type: prometheus
    access: proxy
    url: http://prometheus:9090
    isDefault: true
  - name: Loki
    type: loki
    access: proxy
    url: http://loki:3100
  - name: Tempo
    type: tempo
    access: proxy
    url: http://tempo:3200
    jsonData:
      httpMethod: GET
      serviceMap:
        datasourceUid: prometheus
``
